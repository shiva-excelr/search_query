         divs.forEach(div => {
            div.addEventListener("click", function () {
               // Reset all divs to their original state and background color
               divs.forEach(innerDiv => {
                  innerDiv.setAttribute('data-headlessui-state', 'false');
                  innerDiv.classList.remove('bg-white', 'lg:bg-white/10', 'lg:ring-1', 'lg:ring-inset', 'lg:ring-white/10');
                  innerDiv.style.backgroundColor = '';  // Reset background color
                  const button = innerDiv.querySelector('button');
                  if (button) {
                     button.classList.remove('text-blue-600', 'lg:text-white');
                     button.classList.add('text-blue-100', 'lg:text-white');
                  }
               });

               // Set the clicked div to selected and change its background color
               div.setAttribute('data-headlessui-state', 'selected');
               div.classList.add('bg-white', 'lg:bg-white/10', 'lg:ring-1', 'lg:ring-inset', 'lg:ring-white/10');

               if (clickedButton) {
                  clickedButton.classList.remove('text-blue-100', 'lg:text-white');
                  clickedButton.classList.add('text-blue-600', 'lg:text-white');
               }
            });
         });
      });

   function toggleMenu() {
      var menuIcon = document.getElementById('menuIcon');
      var menuClosed = document.getElementById('menuClosed');
      var menuOpen = document.getElementById('menuOpen');

      var isExpanded = menuIcon.getAttribute('aria-expanded') === 'true';

      // Toggle the aria-expanded attribute
      menuIcon.setAttribute('aria-expanded', !isExpanded);

      // Toggle the display of the SVG paths
      menuClosed.style.display = isExpanded ? 'block' : 'none';
      menuOpen.style.display = isExpanded ? 'none' : 'block';
      menu.classList.toggle('hidden');

      if (!isExpanded) {
        document.addEventListener('click', closeMenuOutside);
    } else {
        document.removeEventListener('click', closeMenuOutside);
    }
   }
   function closeMenuOutside(event) {
    var menuIcon = document.getElementById('menuIcon');
    var menuOpen = document.getElementById('menuOpen');
    var menuClosed = document.getElementById('menuClosed');

    if (event.target !== menuIcon && event.target !== menuOpen && event.target !== menuClosed) {
        toggleMenu();
    }
